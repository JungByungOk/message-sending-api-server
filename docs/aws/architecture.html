CDK 배포 명령어
# 최초 배포 — CDK 부트스트랩 (계정당 1회)
cdk bootstrap aws://123456789012/ap-northeast-2
# 스택 합성 (CloudFormation 템플릿 생성)
cdk synth
# 전체 배포
cdk deploy --all
# IP Whitelist 지정 배포
cdk deploy --context allowedIp=203.0.113.10/32
# Callback Secret 설정 (배포 후 별도 실행 — SecureString 타입)
aws ssm put-parameter \
--name /ems/callback_secret \
--value "your-secret-here" \
--type SecureString --overwrite
# 스택 삭제 (테스트 환경 정리)
cdk destroy --all
